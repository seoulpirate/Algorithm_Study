#include <iostream>
using namespace std;

pair<int, int> calc(int n){
  int t5=0, t2=0;
  for(int i=5; i<=n; i*=5){
    if(n%i == 0)
      t5+=1;
  }
  for(int i=2; i<=n; i*=2){
    if(n%i == 0)
      t2+=1;
  }
  //cout << t5 << ' ' << t2 << '\n';
  return make_pair(t2, t5);
}

int main() {

  //Please Enter Your Code Here
  int n, r;
  cin >> n >> r;
  
  int factorial_1 = n;
  int a_num2 = 0;
  int a_num5 = 0;
  for(int i=1; i<=factorial_1; i++){
    if(i%2==0 || i%5==0){
      auto p = calc(i);
      a_num2 += p.first;
      a_num5 += p.second;
    }
  }
  
  int factorial_2 = r;
  int b_num2 = 0;
  int b_num5 = 0;
  for(int i=1; i<=factorial_2; i++){
    if(i%2==0 || i%5==0){
      auto p = calc(i);
      b_num2 += p.first;
      b_num5 += p.second;
    }
  }
  
  int factorial_3 = (n-r);
  int c_num2 = 0;
  int c_num5 = 0;
  for(int i=1; i<=factorial_3; i++){
    if(i%2==0 || i%5==0){
      auto p = calc(i);
      c_num2 += p.first;
      c_num5 += p.second;
    }
  }
  
  int d_num2 = b_num2 + c_num2;
  int d_num5 = b_num5 + c_num5;
  
  int res2 = a_num2 - d_num2;
  int res5 = a_num5 - d_num5;
  
  //cout << res2 << ' ' << res5 << '\n';
  cout << (res2 < res5 ? res2 : res5) << '\n';
  return 0;
}
