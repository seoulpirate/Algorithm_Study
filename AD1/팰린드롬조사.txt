#include <iostream>
#include <string>
using namespace std;
string a;

int go(int left, int right){
  if(left == right) {
    return 1;
  }
  if(left+1 == right){
    if(a[left] == a[right])
      return 1;
    else
      return 0;
  }
  int res = 0;
  if(a[left] == a[right]){
    res = go(left+1, right-1);
  }
  return res;
}

  cin >> a;
  cout << (go(0, a.length()-1)!=0?"YES":"NO") << '\n';
  return 0;
}
