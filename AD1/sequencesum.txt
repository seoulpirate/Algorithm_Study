int main() {

  //Please Enter Your Code Here
  int n; cin >> n;
  for(int i=0; i<n; i++)
    for(int j=0; j<n; j++)
      cin >> s[i][j];
      
  int num = (s[0][1]+s[0][2]-s[1][2])/2;
  cout << num << ' ';
  for (int i = 1; i < n; i++){
    num = s[i-1][i] - num;

  }

}
