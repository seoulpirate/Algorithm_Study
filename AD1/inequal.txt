bool check(vector<int> &a, vector<char> &oper){
  for(int i=0; i<oper.size(); i++){
    if(oper[i] == '<'){
      if(a[i] > a[i+1])
        return false;
    }else if(oper[i] == '>'){
      if(a[i] < a[i+1])
        return false;
    }
  }
  return true;
}
