#include <iostream>
using namespace std;

int a[100001];
bool c[100001];

int main() {

  //Please Enter Your Code Here
  int n; cin >> n;
  for(int i=0; i<n; i++)
    cin >> a[i];
 
  int left = 0, right = 0; // left: base, right: length
  int ans = 0;
  while(left+right < n){
    if(!c[a[left+right]]){
      // 아직 방문안했어.
      c[a[left+right]] = true;
      right++;
      if(ans < right) 
        ans = right;
    }else{
      // 가보니 방문했어.
      c[a[left]] = false;
      left++;
      right--;
    }
  }
  
  cout << ans << '\n';
  return 0;
}
