#include <iostream>
#include <vector>
using namespace std;
const long long mod = 1000000007LL;
typedef long long ll;
vector<pair<ll, ll>> a[100001];
ll ans = 0;
int go(ll now, ll prev){
  ll ret = 1;
  for(auto next : a[now]){
    if(next.first == prev) continue;
    // now <-> next 경로만 저장
    ll tmp = (go(next.first, now) * next.second) % mod;
    ans = (ans + tmp*ret) % mod;
    // next1 <-> now <-> next2 경로 저장
    ret = (ret+tmp) % mod;
  }
  return ret;
}

int main() {

  //Please Enter Your Code Here
  int n; cin >> n;
  for(int i=0; i<n-1; i++){
    ll x, y, z; cin >> x >> y >> z;
    a[x].push_back({y, z});
    a[y].push_back({x, z});
  }
  if(n==1) {
    cout << 0 << '\n';
    return 0;
  }
  go(1, -1);
  cout << ans << '\n';

  return 0;
}
