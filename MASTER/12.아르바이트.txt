#include <iostream>
#include <cstring>
#include <tuple>
#include <algorithm>
using namespace std;
const int MAX = 100000;
bool c[MAX+1];
int t[MAX+1];
int p[MAX+1];
int n;
int d[MAX+1];
int main(){
  cin >> n;
  for(int i=1; i<=n; i++){
    int start; cin >> start;
    cin >> t[start] >> p[start];
    c[start]=true;
    d[start]=p[start];
  }
  int ans = 0;
  for(int i=2; i<=MAX; i++){
    if(c[i]){
      for(int j=1; j<i; j++){
        if(c[j]){
          if(t[j]<=i){
            d[i] = max(p[i] + d[j], d[i]);
          }
        }
      }
    }
  }
  for(int i=1; i<=MAX; i++)
  if(d[i]>ans) ans = d[i];
 cout << ans << '\n';
  
  return 0;
}
