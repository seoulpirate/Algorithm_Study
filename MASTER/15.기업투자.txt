#include <iostream>
#include <algorithm>
using namespace std;
int a[21][301];
int d[21][301];
// d[i][j] : i번째 기업까지 j만원을 투자해 얻는 최대 이익
// d[i][j] = max(d[i][j], d[i-1][j-k]+a[i][k]) (0<=k<=j)

int main() {

  //Please Enter Your Code Here
  int n, m; cin >> n >> m;
  for(int i=1; i<=n; i++){
    int money; cin >> money;
    for(int j=1; j<=m; j++)
      cin >> a[j][money];
  }
  
  for(int i=1; i<=m; i++){
    for(int j=0; j<=n; j++){
      for(int k=0; k<=j; k++){
        d[i][j] = max(d[i][j], d[i-1][j-k] + a[i][k]);
      }
    }
  }
  cout << d[m][n] << '\n';
  return 0;
}
