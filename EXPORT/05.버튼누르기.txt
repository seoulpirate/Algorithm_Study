#include <iostream>
#include <algorithm>
using namespace std;
/*
d[n][1] : max(d[n-1][2], d[n-1][3]) + a[n][1];
*/
int a[100001][3];
int d[100001][3];
int main() {

  //Please Enter Your Code Here
  int n; cin >> n;
  for(int i=1; i<=n; i++){
    cin >> a[i][0] >> a[i][1] >> a[i][2];
  }
  
  d[1][0] = a[1][0];
  d[1][1] = a[1][1];
  d[1][2] = a[1][2];
  
  for(int i=2; i<=n; i++){
    d[i][0] = max(d[i-1][1], d[i-1][2]) + a[i][0];
    d[i][1] = max(d[i-1][0], d[i-1][2]) + a[i][1];
    d[i][2] = max(d[i-1][0], d[i-1][1]) + a[i][2];
  }
  
  cout << max(d[n][0], max(d[n][1], d[n][2])) << '\n';

  return 0;
}
