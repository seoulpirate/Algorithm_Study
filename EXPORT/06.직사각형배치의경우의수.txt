#include <iostream>
using namespace std;

const int mod = 1000007;
int d[100];

int main() {

  //Please Enter Your Code Here
  int n; cin >> n;
  // d[i] = d[i-1] + d[i-2];
  // ë‹¨, d[1]:1, d[2]:2.
  d[1] = 1, d[2] = 2;
  for(int i=3; i<=n; i++){
    d[i] = d[i-1]+d[i-2];
    d[i] %= mod;
  }

  cout << d[n] << '\n';
  return 0;
}
