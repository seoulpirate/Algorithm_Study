#include <iostream>
#include <algorithm>
using namespace std;
int d[100001];
int a[100001];

int main() {

  //Please Enter Your Code Here

  int n; cin >> n;
  for(int i=1; i<=n; i++)
    cin >> a[i];
  
  d[1] = a[1];
  d[2] = a[1] + a[2];
  d[3] = max(a[3]+a[1], a[3]+a[2]);
  
  for(int i=4; i<=n; i++)
    d[i] = max(d[i-1], max(a[i] + d[i-2], a[i]+a[i-1]+d[i-3]));
  
  cout << d[n] << '\n';
  
  return 0;
}
